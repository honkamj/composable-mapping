 30class DiagonalAffineMatrixDefinition(BaseTensorLikeWrapper):
 31    """Definition of a diagonal affine matrix
 32
 33    Arguments:
 34        diagonal: Tensor with shape ([*batch_shape, ]diagonal_length[, *spatial_shape]),
 35            if None, corresponds to all ones. Can also be given as a sequence of numbers
 36            or a single number.
 37        translation: Tensor with shape ([*batch_shape, ]n_output_dims[, *spatial_shape]),
 38            if None, corresponds to all zeros. Can also be given as a sequence of numbers
 39            or a single number.
 40        matrix_shape: Shape of the target affine transformation matrix
 41            ([*batch_shape, ]n_output_dims + 1, n_input_dims + 1[, *spatial_shape])
 42        dtype: Data type of the transformation matrix, needed only if no
 43            diagonal or translation is given.
 44        device: Device of the transformation matrix, needed only if no
 45            diagonal or translation is given.
 46    """
 47
 48    def __init__(
 49        self,
 50        diagonal: Optional[Union[Tensor, Sequence[Number], Number]] = None,
 51        translation: Optional[Union[Tensor, Sequence[Number], Number]] = None,
 52        matrix_shape: Optional[Sequence[int]] = None,
 53        dtype: Optional[torch_dtype] = None,
 54        device: Optional[torch_device] = None,
 55    ) -> None:
 56        self._dtype, self._device = self._handle_dtype_and_device(
 57            diagonal, translation, dtype, device
 58        )
 59        if isinstance(diagonal, (float, int)):
 60            diagonal = ones(1, dtype=self._dtype, device=self._device) * diagonal
 61        elif not isinstance(diagonal, Tensor) and diagonal is not None:
 62            diagonal = tensor(diagonal, dtype=self._dtype, device=torch_device("cpu"))
 63            if self._device.type != "cpu":
 64                diagonal = diagonal.to(device=self._device, non_blocking=True)
 65        if isinstance(translation, (float, int)):
 66            translation = ones(1, dtype=self._dtype, device=self._device) * translation
 67        elif not isinstance(translation, Tensor) and translation is not None:
 68            translation = tensor(translation, dtype=self._dtype, device=torch_device("cpu"))
 69            if self._device.type != "cpu":
 70                translation = translation.to(device=self._device, non_blocking=True)
 71        if diagonal is not None and diagonal.ndim == 0:
 72            diagonal = diagonal.unsqueeze(0)
 73        if translation is not None and translation.ndim == 0:
 74            translation = translation.unsqueeze(0)
 75        batch_shape, channels_shape, spatial_shape = self._infer_diagonal_matrix_shape_in_parts(
 76            diagonal, translation, matrix_shape
 77        )
 78        diagonal_length = min(channels_shape[0] - 1, channels_shape[1] - 1)
 79        if diagonal is not None:
 80            diagonal = broadcast_to_in_parts(
 81                diagonal,
 82                channels_shape=(diagonal_length,),
 83            )
 84        if translation is not None:
 85            translation = broadcast_to_in_parts(
 86                translation,
 87                channels_shape=(channels_shape[0] - 1,),
 88            )
 89        self._broadcasted_shape = batch_shape + channels_shape + spatial_shape
 90        self._matrix_shape = matrix_shape
 91        self._diagonal = diagonal
 92        self._translation = translation

178    def as_matrix(self) -> Tensor:
179        """Generate the diagonal affine matrix"""
180        batch_shape, channels_shape, spatial_shape = split_shape(self.shape, n_channel_dims=2)
181        matrix = generate_scale_matrix(self.generate_diagonal())
182        matrix = broadcast_to_in_parts(
183            matrix,
184            batch_shape=batch_shape,
185            spatial_shape=spatial_shape,
186            n_channel_dims=2,
187        )
188        if self._translation is not None:
189            translation = broadcast_to_in_parts(
190                self._translation,
191                batch_shape=batch_shape,
192                spatial_shape=spatial_shape,
193            )
194            matrix = embed_matrix(matrix, (channels_shape[0] - 1, channels_shape[1] - 1))
195            matrix_channel_dims = get_channel_dims(matrix.ndim, n_channel_dims=2)
196            matrix = cat(
197                (matrix, translation.unsqueeze(matrix_channel_dims[1])),
198                dim=matrix_channel_dims[1],
199            )
200        matrix = embed_matrix(matrix, channels_shape)
201        return matrix

203    def generate_diagonal(self) -> Tensor:
204        """Generate the diagonal"""
205        batch_shape, channels_shape, spatial_shape = split_shape(self.shape, n_channel_dims=2)
206        if self._diagonal is None:
207            diagonal_length = min(channels_shape[0] - 1, channels_shape[1] - 1)
208            diagonal = ones(
209                diagonal_length,
210                dtype=self._dtype,
211                device=self._device,
212            )
213        else:
214            diagonal = self._diagonal
215        return broadcast_to_in_parts(
216            diagonal,
217            batch_shape=batch_shape,
218            spatial_shape=spatial_shape,
219            n_channel_dims=1,
220        )

224        batch_shape, channels_shape, spatial_shape = split_shape(self.shape, n_channel_dims=2)
225        if self._translation is None:
226            translation = ones(
227                channels_shape[0] - 1,
228                dtype=self._dtype,

231        else:
232            translation = self._translation

234            translation,

240    @staticmethod
241    def _infer_diagonal_matrix_shape_in_parts(
242        diagonal: Optional[Tensor],
243        translation: Optional[Tensor],
244        matrix_shape: Optional[Sequence[int]],
245    ) -> Tuple[Tuple[int, ...], Tuple[int, ...], Tuple[int, ...]]:
246        """Get the shape of a diagonal matrix in parts"""
247        if matrix_shape is None:
248            if diagonal is not None and translation is not None:
249                channels_shape: Tuple[int, ...] = (
250                    get_channels_shape(translation.shape, n_channel_dims=1)[0] + 1,
251                    get_channels_shape(diagonal.shape, n_channel_dims=1)[0] + 1,
252                )
253            elif diagonal is not None:
254                channels_shape = (
255                    get_channels_shape(diagonal.shape, n_channel_dims=1)[0] + 1,
256                    get_channels_shape(diagonal.shape, n_channel_dims=1)[0] + 1,
257                )
258            elif translation is not None:
259                channels_shape = (
260                    get_channels_shape(translation.shape, n_channel_dims=1)[0] + 1,
261                    get_channels_shape(translation.shape, n_channel_dims=1)[0] + 1,
262                )
263            else:
264                raise ValueError(
265                    "At least one of diagonal, translation, and matrix_shape must be given."
266                )
267        else:
268            channels_shape = get_channels_shape(matrix_shape, n_channel_dims=2)
269        if diagonal is not None:
270            diagonal_length = min(channels_shape[0] - 1, channels_shape[1] - 1)
271            diagonal = broadcast_to_in_parts(
272                diagonal,
273                channels_shape=(diagonal_length,),
274            )
275        if translation is not None:
276            translation = broadcast_to_in_parts(
277                translation,
278                channels_shape=(channels_shape[0] - 1,),
279            )
280        batch_size, _, spatial_shape = broadcast_optional_shapes_in_parts_splitted(
281            diagonal.shape if diagonal is not None else None,
282            translation.shape if translation is not None else None,
283            matrix_shape,
284            n_channel_dims=(1, 1, 2),
285            broadcast_channels=False,
286        )
287        assert batch_size is not None and spatial_shape is not None
288        return batch_size, channels_shape, spatial_shape

290    @staticmethod
291    def _handle_dtype_and_device(
292        diagonal: Optional[Union[Tensor, Sequence[Number], Number]],
293        translation: Optional[Union[Tensor, Sequence[Number], Number]],
294        dtype: Optional[torch_dtype],
295        device: Optional[torch_device],
296    ) -> Tuple[torch_dtype, torch_device]:
297        if diagonal is not None and isinstance(diagonal, Tensor):
298            if device is not None and device != diagonal.device:
299                raise ValueError(
300                    "Device mismatch, note that the device is needed only if no diagonal or "
301                    "translation is given as tensor."
302                )
303            if dtype is not None and dtype != diagonal.dtype:
304                raise ValueError(
305                    "Dtype mismatch, note that the dtype is needed only if no diagonal or "
306                    "translation is given as tensor."
307                )
308            dtype = diagonal.dtype
309            device = diagonal.device
310            if translation is not None and isinstance(translation, Tensor):
311                if translation.device != device:
312                    raise ValueError("Device mismatch between diagonal and translation.")
313                if translation.dtype != dtype:
314                    raise ValueError("Dtype mismatch between diagonal and translation.")
315        elif translation is not None and isinstance(translation, Tensor):
316            if device is not None and device != translation.device:
317                raise ValueError(
318                    "Device mismatch, note that the device is needed only if no diagonal or "
319                    "translation is given."
320                )
321            if dtype is not None and dtype != translation.dtype:
322                raise ValueError(
323                    "Dtype mismatch, note that the dtype is needed only if no diagonal or "
324                    "translation is given."
325                )
326            dtype = translation.dtype
327            device = translation.device
328        dtype = get_default_dtype() if dtype is None else dtype
329        device = get_default_device() if device is None else device
330        return dtype, device

343def transform_values_with_diagonal_affine_matrix(
344    matrix_definition: DiagonalAffineMatrixDefinition,
345    values: Tensor,
346    n_channel_dims: int = 1,
347) -> Tensor:
348    """Transform values with a diagonal affine matrix
349
350    Args:
351        matrix_definition: Definition of the diagonal affine matrix.
352        values: Values to transform.
353        n_channel_dims: Number of channel dimensions in the values tensor.
354
355    Returns:
356        Values transformed with the diagonal affine matrix.
357    """
358    affine_channels_shape = get_channels_shape(matrix_definition.shape, n_channel_dims=2)
359    n_input_dims, n_output_dims = (affine_channels_shape[1] - 1, affine_channels_shape[0] - 1)
360    if get_channels_shape(values.shape, n_channel_dims=n_channel_dims)[-1] != n_input_dims:
361        raise ValueError("The diagonal matrix does not match the number of dimensions.")
362    diagonal_length = min(affine_channels_shape[0] - 1, affine_channels_shape[1] - 1)
363    values_channel_dim = get_channel_dims(values.ndim, n_channel_dims=n_channel_dims)[-1]
364    if diagonal_length < n_input_dims:
365        values = values.narrow(values_channel_dim, 0, diagonal_length)
366    if matrix_definition.diagonal is not None:
367        diagonal, values = broadcast_tensors_in_parts(
368            matrix_definition.generate_diagonal(), values, n_channel_dims=(1, n_channel_dims)
369        )
370        values = values * diagonal
371
372    if diagonal_length < n_output_dims:
373        values = _pad_dimension_with_zeros(values, n_output_dims, values_channel_dim)
374    if matrix_definition.translation is not None:
375        translation, values = broadcast_tensors_in_parts(
376            matrix_definition.generate_translation(), values, n_channel_dims=(1, n_channel_dims)
377        )
378        values = values + translation
379    return values

382def invert_diagonal_affine_matrix(
383    matrix_definition: DiagonalAffineMatrixDefinition,
384) -> DiagonalAffineMatrixDefinition:
385    """Invert a diagonal affine matrix
386
387    Args:
388        matrix_definition: Definition of the diagonal affine matrix.
389
390    Returns:
391        Definition of the inverted diagonal affine matrix.
392    """
393    affine_channels_shape = get_channels_shape(matrix_definition.shape, n_channel_dims=2)
394    if affine_channels_shape[0] != affine_channels_shape[1]:
395        raise ValueError("The diagonal matrix must be square.")
396    if matrix_definition.diagonal is None:
397        diagonal: Optional[Tensor] = None
398    else:
399        diagonal = 1 / matrix_definition.diagonal
400    if matrix_definition.translation is None:
401        translation: Optional[Tensor] = None
402    else:
403        translation = -matrix_definition.translation
404        if diagonal is not None:
405            diagonal, translation = broadcast_tensors_in_parts(
406                diagonal, translation, n_channel_dims=1
407            )
408            translation = translation * diagonal
409    return DiagonalAffineMatrixDefinition(
410        diagonal,
411        translation,
412        matrix_definition.shape,
413        dtype=matrix_definition.dtype,
414        device=matrix_definition.device,
415    )

433        translation = None
434    if matrix_definition.diagonal is not None:

439            diagonal_length, dtype=matrix_definition.dtype, device=matrix_definition.device

442        diagonal,
443        translation,
444        matrix_shape=matrix_definition.shape,
445        dtype=matrix_definition.dtype,
446        device=matrix_definition.device,

450def is_identity_diagonal_affine_matrix(
451    matrix_definition: DiagonalAffineMatrixDefinition,
452) -> bool:
453    """Is the diagonal matrix an identity matrix.
454
455    Args:
456        matrix_definition: Definition of the diagonal affine matrix.
457
458    Returns:
459        Whether the diagonal matrix is an identity matrix.
460    """
461    if matrix_definition.diagonal is not None:
462        if not allclose(
463            matrix_definition.diagonal,
464            ones(1, dtype=matrix_definition.dtype, device=matrix_definition.device),
465            atol=IDENTITY_MATRIX_TOLERANCE,
466        ):
467            return False
468    if matrix_definition.translation is not None:
469        if not allclose(
470            matrix_definition.translation,
471            zeros(1, dtype=matrix_definition.dtype, device=matrix_definition.device),
472            atol=IDENTITY_MATRIX_TOLERANCE,
473        ):
474            return False
475    return True

514def compose_diagonal_affine_matrices(
515    matrix_definition_1: DiagonalAffineMatrixDefinition,
516    matrix_definition_2: DiagonalAffineMatrixDefinition,
517) -> DiagonalAffineMatrixDefinition:
518    """Compose two diagonal affine matrices.
519
520    Args:
521        matrix_definition_1: Definition of the first diagonal affine matrix.
522        matrix_definition_2: Definition of the second diagonal affine matrix.
523
524    Returns:
525        Definition of the composed diagonal affine matrix.
526    """
527    batch_shape_1, channels_shape_1, spatial_shape_1 = split_shape(
528        matrix_definition_1.shape, n_channel_dims=2
529    )
530    batch_shape_2, channels_shape_2, spatial_shape_2 = split_shape(
531        matrix_definition_2.shape, n_channel_dims=2
532    )
533    diagonal_length_1 = min(channels_shape_1[0] - 1, channels_shape_1[1] - 1)
534    diagonal_length_2 = min(channels_shape_2[0] - 1, channels_shape_2[1] - 1)
535    shared_diagonal_length = min(diagonal_length_1, diagonal_length_2)
536    batch_shape = broadcast_shapes(batch_shape_1, batch_shape_2)
537    spatial_shape = broadcast_shapes(spatial_shape_1, spatial_shape_2)
538    if channels_shape_1[1] != channels_shape_2[0]:
539        raise ValueError("The matrices are not compatible.")
540    channels_shape = (channels_shape_1[0], channels_shape_2[1])
541    diagonal_length = min(channels_shape[0] - 1, channels_shape[1] - 1)
542
543    diagonal_1 = matrix_definition_1.diagonal
544    diagonal_2 = matrix_definition_2.diagonal
545    if diagonal_1 is not None and shared_diagonal_length < diagonal_length_1:
546        channel_dim = get_channel_dims(diagonal_1.ndim, n_channel_dims=1)[0]
547        diagonal_1 = diagonal_1.narrow(channel_dim, 0, diagonal_length)
548    if diagonal_2 is not None and shared_diagonal_length < diagonal_length_2:
549        channel_dim = get_channel_dims(diagonal_2.ndim, n_channel_dims=1)[0]
550        diagonal_2 = diagonal_2.narrow(channel_dim, 0, diagonal_length)
551
552    if diagonal_1 is not None and diagonal_2 is not None:
553        diagonal_1, diagonal_2 = broadcast_tensors_in_parts(
554            diagonal_1, diagonal_2, n_channel_dims=1
555        )
556        diagonal: Optional[Tensor] = diagonal_1 * diagonal_2
557    elif diagonal_1 is not None:
558        diagonal = diagonal_1
559    elif diagonal_2 is not None:
560        diagonal = diagonal_2
561    else:
562        diagonal = None
563    if diagonal is not None:
564        channel_dim = get_channel_dims(diagonal.ndim, n_channel_dims=1)[0]
565        diagonal = _pad_dimension_with_zeros(diagonal, diagonal_length, channel_dim)
566
567    if matrix_definition_2.translation is not None:
568        channel_dim = get_channel_dims(matrix_definition_2.translation.ndim, n_channel_dims=1)[0]
569        translation_2 = matrix_definition_2.translation.narrow(channel_dim, 0, diagonal_length_1)
570        if diagonal_1 is not None:
571            diagonal_1, translation_2 = broadcast_tensors_in_parts(
572                diagonal_1, translation_2, n_channel_dims=1
573            )
574            translation_2 = translation_2 * diagonal_1
575        channel_dim = get_channel_dims(translation_2.ndim, n_channel_dims=1)[0]
576        translation_2 = _pad_dimension_with_zeros(translation_2, channels_shape[0] - 1, channel_dim)
577    else:
578        translation_2 = None
579
580    if matrix_definition_1.translation is not None and translation_2 is not None:
581        translation_1, translation_2 = broadcast_tensors_in_parts(
582            matrix_definition_1.translation, translation_2, n_channel_dims=1
583        )
584        translation: Optional[Tensor] = translation_2 + translation_1
585    elif matrix_definition_1.translation is not None:
586        translation = matrix_definition_1.translation
587    elif translation_2 is not None:
588        translation = translation_2
589    else:
590        translation = None
591
592    return DiagonalAffineMatrixDefinition(
593        diagonal,
594        translation,
595        matrix_shape=batch_shape + channels_shape + spatial_shape,
596        dtype=matrix_definition_1.dtype,
597        device=matrix_definition_1.device,
598    )

601def add_diagonal_affine_matrices(
602    matrix_definition_1: DiagonalAffineMatrixDefinition,
603    matrix_definition_2: DiagonalAffineMatrixDefinition,
604) -> DiagonalAffineMatrixDefinition:
605    """Add two diagonal affine matrices.
606
607    Args:
608        matrix_definition_1: Definition of the first diagonal affine matrix.
609        matrix_definition_2: Definition of the second diagonal affine matrix.
610
611    Returns:
612        Definition of the sum of the two diagonal affine matrices.
613    """
614    matrix_shape = broadcast_shapes_in_parts_to_single_shape(
615        matrix_definition_1.shape, matrix_definition_2.shape, n_channel_dims=2
616    )
617    if get_channels_shape(matrix_definition_1.shape, n_channel_dims=2) != get_channels_shape(
618        matrix_definition_2.shape, n_channel_dims=2
619    ):
620        raise ValueError("The matrices are not compatible.")
621
622    if matrix_definition_1.diagonal is None and matrix_definition_2.diagonal is None:
623        diagonal: Optional[Tensor] = None
624    else:
625        diagonal_1, diagonal_2 = broadcast_tensors_in_parts(
